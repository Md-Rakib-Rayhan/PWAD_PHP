CREATE TABLE products (

    product_name VARCHAR (100) NOT NULL,
    price FLOAT(10,2) NOT NULL,
    quantity SMALLINT NULL
)

--------------

SELECT `productID`, `product_name`, `price` FROM `products` WHERE price<=10;
SELECT `productID`, `product_name`, `price` FROM `products` WHERE price BETWEEN 30 AND 50;   ->Mean price>=30 AND price<=50;


Sorting/Order by
--------------
SELECT `productID`, `product_name`, `price` FROM `products` WHERE price BETWEEN 30 AND 50 ORDER BY product_name;
SELECT `productID`, `product_name`, `price` FROM `products` WHERE price BETWEEN 30 AND 50 ORDER BY product_name ASC;
SELECT `productID`, `product_name`, `price` FROM `products` WHERE price BETWEEN 30 AND 50 ORDER BY product_name DESC;
